22 struct EventSelectionQaPerBc {

24  OutputObj<TH1F> hFiredClasses{TH1F("hFiredClasses", "", 100, -0.5, 99.5)};
25  OutputObj<TH1F> hFiredAliases{TH1F("hFiredAliases", "", kNaliases, -0.5, kNaliases - 0.5)};

27  {

33  void process(soa::Join<aod::BCs, aod::Run2BCInfos, aod::BcSels>::iterator const& bc)
34  {
35  // Fill fired trigger classes
36  uint64_t triggerMask = bc.triggerMask();
37  uint64_t triggerMaskNext50 = bc.triggerMaskNext50();
38  for (int i = 0; i < 50; i++) {
39  if (triggerMask & 1ull << i) {
40  hFiredClasses->Fill(i);
41  }
42  if (triggerMaskNext50 & 1ull << i) {
43  hFiredClasses->Fill(i + 50);
44  }
45  }
46 
47  // Fill fired aliases
48  for (int i = 0; i < kNaliases; i++) {
49  if (bc.alias()[i]) {
50  hFiredAliases->Fill(i);
51  }
52  }
53  }
54 };
