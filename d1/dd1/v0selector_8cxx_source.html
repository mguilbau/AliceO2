107  void process(soa::Join<aod::Collisions, aod::EvSels>::iterator const& collision, aod::V0Datas const& fullV0s, FullTracksExt const& tracks)
108  {
109 
110  //if (!collision.alias()[kINT7]) {
111  // return;
112  //}
113  //if (!collision.sel7()) {
114  // return;
115  //}
116 
117  for (auto& v0 : fullV0s) {
118  if (v0.negTrack_as<FullTracksExt>().tpcChi2NCl() > 4.0 || v0.posTrack_as<FullTracksExt>().tpcChi2NCl() > 4.0) {
119  continue;
120  }
121  if (v0.negTrack_as<FullTracksExt>().tpcNClsCrossedRows() < 70 || v0.posTrack_as<FullTracksExt>().tpcNClsCrossedRows() < 70) {
122  continue;
123  }
124  if (!(v0.negTrack_as<FullTracksExt>().trackType() & o2::aod::track::TPCrefit) || !(v0.posTrack_as<FullTracksExt>().trackType() & o2::aod::track::TPCrefit)) {
125  continue;
126  }
127 
128  registry.fill(HIST("hV0Pt"), v0.pt());
129  registry.fill(HIST("hV0EtaPhi"), v0.phi(), v0.eta());
130 
131  registry.fill(HIST("hV0Radius"), v0.v0radius());
132  registry.fill(HIST("hV0CosPA"), v0.v0cosPA(collision.posX(), collision.posY(), collision.posZ()));
133  registry.fill(HIST("hDCAPosToPV"), v0.dcapostopv());
134  registry.fill(HIST("hDCANegToPV"), v0.dcanegtopv());
135  registry.fill(HIST("hDCAV0Dau"), v0.dcaV0daughters());
136  registry.fill(HIST("hV0APplot"), v0.alpha(), v0.qtarm());
137 
138  registry.fill(HIST("hMassK0S"), v0.mK0Short());
139  registry.fill(HIST("hMassLambda"), v0.mLambda());
140  registry.fill(HIST("hMassAntiLambda"), v0.mAntiLambda());
141 
142  registry.fill(HIST("h2TPCdEdx_Pin_Neg"), v0.negTrack_as<FullTracksExt>().tpcInnerParam(), v0.negTrack_as<FullTracksExt>().tpcSignal());
143  registry.fill(HIST("h2TPCdEdx_Pin_Pos"), v0.posTrack_as<FullTracksExt>().tpcInnerParam(), v0.posTrack_as<FullTracksExt>().tpcSignal());
144 
145  registry.fill(HIST("h2TOFbeta_Pin_Neg"), v0.negTrack_as<FullTracksExt>().tpcInnerParam(), v0.negTrack_as<FullTracksExt>().beta());
146  registry.fill(HIST("h2TOFbeta_Pin_Pos"), v0.posTrack_as<FullTracksExt>().tpcInnerParam(), v0.posTrack_as<FullTracksExt>().beta());
147 
148  if ((70.0 < v0.negTrack_as<FullTracksExt>().tpcSignal() && v0.negTrack_as<FullTracksExt>().tpcSignal() < 90) && (70.0 < v0.posTrack_as<FullTracksExt>().tpcSignal() && v0.posTrack_as<FullTracksExt>().tpcSignal() < 90)) {
149  registry.fill(HIST("hMassGamma"), v0.mGamma());
150  registry.fill(HIST("hV0APplot_Gamma"), v0.alpha(), v0.qtarm());
151  registry.fill(HIST("h2MassGammaR"), v0.v0radius(), v0.mGamma());
152  }
153 
154  if (TMath::Abs(v0.alpha()) < 0.4 && v0.qtarm() < 0.03 && v0.mGamma() < 0.01) { //photon conversion
155  registry.fill(HIST("hMassGamma_AP"), v0.mGamma());
156  registry.fill(HIST("h2TPCdEdx_Pin_El_minus"), v0.negTrack_as<FullTracksExt>().tpcInnerParam(), v0.negTrack_as<FullTracksExt>().tpcSignal());
157  registry.fill(HIST("h2TPCdEdx_Pin_El_plus"), v0.posTrack_as<FullTracksExt>().tpcInnerParam(), v0.posTrack_as<FullTracksExt>().tpcSignal());
158  registry.fill(HIST("h2TOFbeta_Pin_El_minus"), v0.negTrack_as<FullTracksExt>().tpcInnerParam(), v0.negTrack_as<FullTracksExt>().beta());
159  registry.fill(HIST("h2TOFbeta_Pin_El_plus"), v0.posTrack_as<FullTracksExt>().tpcInnerParam(), v0.posTrack_as<FullTracksExt>().beta());
160 
161  } else if (TMath::Abs(v0.alpha()) < 0.7 && (0.11 < v0.qtarm() && v0.qtarm() < 0.22) && (0.49 < v0.mK0Short() && v0.mK0Short() < 0.51)) { //K0S-> pi pi
162  registry.fill(HIST("hMassK0S_AP"), v0.mK0Short());
163  registry.fill(HIST("h2TPCdEdx_Pin_Pi_minus"), v0.negTrack_as<FullTracksExt>().tpcInnerParam(), v0.negTrack_as<FullTracksExt>().tpcSignal());
164  registry.fill(HIST("h2TPCdEdx_Pin_Pi_plus"), v0.posTrack_as<FullTracksExt>().tpcInnerParam(), v0.posTrack_as<FullTracksExt>().tpcSignal());
165  registry.fill(HIST("h2TOFbeta_Pin_Pi_minus"), v0.negTrack_as<FullTracksExt>().tpcInnerParam(), v0.negTrack_as<FullTracksExt>().beta());
166  registry.fill(HIST("h2TOFbeta_Pin_Pi_plus"), v0.posTrack_as<FullTracksExt>().tpcInnerParam(), v0.posTrack_as<FullTracksExt>().beta());
167 
168  } else if ((+0.45 < v0.alpha() && v0.alpha() < +0.7) && (0.03 < v0.qtarm() && v0.qtarm() < 0.11) && (1.112 < v0.mLambda() && v0.mLambda() < 1.120)) { //L->p + pi-
169  registry.fill(HIST("hMassLambda_AP"), v0.mLambda());
170  registry.fill(HIST("h2TPCdEdx_Pin_Pi_minus"), v0.negTrack_as<FullTracksExt>().tpcInnerParam(), v0.negTrack_as<FullTracksExt>().tpcSignal());
171  registry.fill(HIST("h2TPCdEdx_Pin_Pr_plus"), v0.posTrack_as<FullTracksExt>().tpcInnerParam(), v0.posTrack_as<FullTracksExt>().tpcSignal());
172  registry.fill(HIST("h2TOFbeta_Pin_Pi_minus"), v0.negTrack_as<FullTracksExt>().tpcInnerParam(), v0.negTrack_as<FullTracksExt>().beta());
173  registry.fill(HIST("h2TOFbeta_Pin_Pr_plus"), v0.posTrack_as<FullTracksExt>().tpcInnerParam(), v0.posTrack_as<FullTracksExt>().beta());
174 
175  } else if ((-0.7 < v0.alpha() && v0.alpha() < -0.45) && (0.03 < v0.qtarm() && v0.qtarm() < 0.11) && (1.112 < v0.mAntiLambda() && v0.mAntiLambda() < 1.120)) { //Lbar -> pbar + pi+
176  registry.fill(HIST("hMassAntiLambda_AP"), v0.mAntiLambda());
177  registry.fill(HIST("h2TPCdEdx_Pin_Pr_minus"), v0.negTrack_as<FullTracksExt>().tpcInnerParam(), v0.negTrack_as<FullTracksExt>().tpcSignal());
178  registry.fill(HIST("h2TPCdEdx_Pin_Pi_plus"), v0.posTrack_as<FullTracksExt>().tpcInnerParam(), v0.posTrack_as<FullTracksExt>().tpcSignal());
179  registry.fill(HIST("h2TOFbeta_Pin_Pr_minus"), v0.negTrack_as<FullTracksExt>().tpcInnerParam(), v0.negTrack_as<FullTracksExt>().beta());
180  registry.fill(HIST("h2TOFbeta_Pin_Pi_plus"), v0.posTrack_as<FullTracksExt>().tpcInnerParam(), v0.posTrack_as<FullTracksExt>().beta());
181  }
182  }
183 
184  //loop for pi0/eta->gamma gamma
185  for (auto& [t1, t2] : combinations(fullV0s, fullV0s)) {
186 
187  if (t1.negTrack_as<FullTracksExt>().tpcSignal() < 70.0 || 90.0 < t1.negTrack_as<FullTracksExt>().tpcSignal()) {
188  continue;
189  }
190  if (t1.posTrack_as<FullTracksExt>().tpcSignal() < 70.0 || 90.0 < t1.posTrack_as<FullTracksExt>().tpcSignal()) {
191  continue;
192  }
193  if (t2.negTrack_as<FullTracksExt>().tpcSignal() < 70.0 || 90.0 < t2.negTrack_as<FullTracksExt>().tpcSignal()) {
194  continue;
195  }
196  if (t2.posTrack_as<FullTracksExt>().tpcSignal() < 70.0 || 90.0 < t2.posTrack_as<FullTracksExt>().tpcSignal()) {
197  continue;
198  }
199  if (t1.negTrack_as<FullTracksExt>().tpcChi2NCl() > 4.0 || t1.posTrack_as<FullTracksExt>().tpcChi2NCl() > 4.0) {
200  continue;
201  }
202  if (t2.negTrack_as<FullTracksExt>().tpcChi2NCl() > 4.0 || t2.posTrack_as<FullTracksExt>().tpcChi2NCl() > 4.0) {
203  continue;
204  }
205 
206  //if(TMath::Abs(t1.alpha()) > 0.35 || TMath::Abs(t2.alpha()) > 0.35) continue;
207  if (t1.qtarm() > 0.03 || t2.qtarm() > 0.03) {
208  continue;
209  }
210  if (t1.mGamma() > 0.01 || t2.mGamma() > 0.01) {
211  continue;
212  }
213  if (t1.v0radius() > 50 || t2.v0radius() > 50) {
214  continue;
215  }
216 
217  //ROOT::Math::PtEtaPhiMVector v1(t1.pt(), t1.eta(), t1.phi(), t1.mGamma());
218  //ROOT::Math::PtEtaPhiMVector v2(t2.pt(), t2.eta(), t2.phi(), t2.mGamma());
219  ROOT::Math::PtEtaPhiMVector v1(t1.pt(), t1.eta(), t1.phi(), 0.0);
220  ROOT::Math::PtEtaPhiMVector v2(t2.pt(), t2.eta(), t2.phi(), 0.0);
221  ROOT::Math::PtEtaPhiMVector v12 = v1 + v2;
222  registry.fill(HIST("h2MggPt"), v12.M(), v12.Pt());
223  }
224  }
225 };
